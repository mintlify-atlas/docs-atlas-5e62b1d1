## Overview

Click-n-Color uses Chrome's runtime messaging API to communicate between the popup, background service worker, and content scripts. This page documents all message types and their data structures.

## Message flow

The extension follows this communication pattern:

1. **Popup → Background**: User clicks "Pick Color" button
2. **Background → Content**: Screenshot data sent to injected content script
3. **Content → Background**: Picked color RGB value sent back
4. **Background → Popup**: Color saved and popup reopened

## Message types

### startColorPick

Sent from the popup to the background service worker to initiate the color picking process.

<ParamField path="type" type="string" required>
  Message identifier
  
  ```javascript
  "startColorPick"
  ```
</ParamField>

#### Request example

```javascript popup.js
chrome.runtime.sendMessage({ type: 'startColorPick' }, (response) => {
  if (chrome.runtime.lastError) {
    console.error('startColorPick message error:', chrome.runtime.lastError.message);
  }
});
```

#### Response

<ResponseField name="status" type="string">
  Indicates whether the color picker started successfully.
  
  Possible values:
  - `"started"` - Color picker initialized successfully
  - `"error"` - Failed to start color picker
</ResponseField>

<ResponseField name="message" type="string">
  Error message if status is "error".
  
  Possible error messages:
  - `"No active tab found."`
  - `"This page is not supported for color picking."`
  - `"Unable to start picker on this page."`
</ResponseField>

#### Background handler

```javascript background.js
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  if (message.type === 'startColorPick') {
    (async () => {
      try {
        const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
        if (!tab || !tab.id) {
          sendResponse({ status: 'error', message: 'No active tab found.' });
          return;
        }

        if (!isTabUrlSupported(tab.url)) {
          sendResponse({ status: 'error', message: 'This page is not supported for color picking.' });
          return;
        }

        const imageUri = await captureVisibleTab(tab.windowId);
        await chrome.scripting.executeScript({
          target: { tabId: tab.id },
          files: ['content.js']
        });

        await sendScreenshotWithRetry(tab.id, imageUri, 6);
        sendResponse({ status: 'started' });
      } catch (error) {
        console.error('startColorPick failed:', error);
        sendResponse({ status: 'error', message: 'Unable to start picker on this page.' });
      }
    })();

    return true;
  }
});
```

### screenshot

Sent from the background service worker to the content script with the captured page screenshot.

<ParamField path="type" type="string" required>
  Message identifier
  
  ```javascript
  "screenshot"
  ```
</ParamField>

<ParamField path="imageUri" type="string" required>
  Base64-encoded PNG data URI of the captured visible tab.
  
  ```javascript
  "data:image/png;base64,iVBORw0KGg..."
  ```
</ParamField>

#### Send example

```javascript background.js
function sendScreenshotWithRetry(tabId, imageUri, retriesLeft = 6) {
  return new Promise((resolve, reject) => {
    const attempt = (remainingRetries) => {
      chrome.tabs.sendMessage(tabId, {
        type: 'screenshot',
        imageUri
      }, (response) => {
        if (chrome.runtime.lastError || !response || response.status !== 'ready') {
          if (remainingRetries > 0) {
            setTimeout(() => attempt(remainingRetries - 1), 120);
            return;
          }

          reject(new Error(chrome.runtime.lastError?.message || 'No response from content script'));
          return;
        }

        resolve();
      });
    };

    attempt(retriesLeft);
  });
}
```

#### Response

<ResponseField name="status" type="string">
  Confirms that the content script received the screenshot.
  
  Value: `"ready"`
</ResponseField>

#### Content script handler

```javascript content.js
chrome.runtime.onMessage.addListener(handleMessage);

handleMessage = (message, sender, sendResponse) => {
  if (message.type === 'screenshot') {
    if (initTimeoutId) {
      clearTimeout(initTimeoutId);
      initTimeoutId = null;
    }

    sendResponse({ status: 'ready' });

    if (isInitializing || isImageReady) {
      return;
    }

    isInitializing = true;
    attachUi();
    document.addEventListener('click', handleClick);

    const img = new Image();
    img.src = message.imageUri;

    img.onload = () => {
      canvas = document.createElement('canvas');
      devicePixelRatio = window.devicePixelRatio || 1;
      
      canvas.width = img.width;
      canvas.height = img.height;
      
      ctx = canvas.getContext('2d', { willReadFrequently: true });
      ctx.drawImage(img, 0, 0);
      isImageReady = true;
      isInitializing = false;

      // Enable color picking
      document.addEventListener('mousemove', handleMouseMove);
    };
  }
};
```

### colorPicked

Sent from the content script to the background service worker when the user clicks to pick a color.

<ParamField path="type" type="string" required>
  Message identifier
  
  ```javascript
  "colorPicked"
  ```
</ParamField>

<ParamField path="color" type="string" required>
  RGB color value in CSS rgb() format.
  
  ```javascript
  "rgb(138, 43, 226)"
  ```
</ParamField>

#### Send example

```javascript content.js
function pickColorAt(clientX, clientY) {
  try {
    const color = getColorAt(clientX, clientY);

    chrome.runtime.sendMessage({
      type: 'colorPicked',
      color: color
    });
  } catch (error) {
    console.error('Error picking color:', error);
  }

  cleanup();
}

function getColorAt(clientX, clientY) {
  const canvasX = Math.floor(clientX * devicePixelRatio);
  const canvasY = Math.floor(clientY * devicePixelRatio);

  const clampedX = Math.max(0, Math.min(canvasX, canvas.width - 1));
  const clampedY = Math.max(0, Math.min(canvasY, canvas.height - 1));

  const pixel = ctx.getImageData(clampedX, clampedY, 1, 1).data;
  return `rgb(${pixel[0]}, ${pixel[1]}, ${pixel[2]})`;
}
```

#### Response

<ResponseField name="status" type="string">
  Confirms that the color was saved successfully.
  
  Value: `"saved"`
</ResponseField>

#### Background handler

```javascript background.js
chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  if (message.type !== 'colorPicked' || typeof message.color !== 'string') {
    return;
  }

  chrome.storage.local.get(['savedColors', 'settings'], (result) => {
    let savedColors = Array.isArray(result.savedColors) ? result.savedColors : [];
    const settings = result.settings && typeof result.settings === 'object' ? result.settings : {};
    const parsedMax = parseInt(settings.maxPaletteSize, 10);
    const maxPaletteSize = Number.isFinite(parsedMax)
      ? Math.min(100, Math.max(10, parsedMax))
      : 50;

    if (!savedColors.includes(message.color)) {
      savedColors.unshift(message.color);
      if (savedColors.length > maxPaletteSize) {
        savedColors = savedColors.slice(0, maxPaletteSize);
      }
    }

    chrome.storage.local.set({
      savedColors,
      lastPickedColor: message.color
    }, () => {
      chrome.action.openPopup().catch(() => {
        chrome.windows.create({
          url: chrome.runtime.getURL('popup.html'),
          type: 'popup',
          width: 560,
          height: 760,
          focused: true
        });
      });

      sendResponse({ status: 'saved' });
    });
  });

  return true;
});
```

## Storage API

Click-n-Color uses `chrome.storage.local` to persist user data across sessions.

### Storage keys

<ParamField path="savedColors" type="array">
  Array of RGB color strings saved by the user.
  
  ```javascript
  ["rgb(138, 43, 226)", "rgb(0, 191, 255)", "rgb(255, 20, 20)"]
  ```
</ParamField>

<ParamField path="lastPickedColor" type="string">
  The most recently picked color in RGB format.
  
  ```javascript
  "rgb(138, 43, 226)"
  ```
</ParamField>

<ParamField path="settings" type="object">
  User preferences for the extension.
  
  <Accordion title="Settings properties">
    <ResponseField name="defaultFormat" type="string">
      Default color format to display. Options: `"rgb"`, `"hex"`, `"hsl"`
    </ResponseField>
    
    <ResponseField name="maxPaletteSize" type="number">
      Maximum number of colors to save in the palette. Range: 10-100, default: 50
    </ResponseField>
    
    <ResponseField name="autoCopy" type="boolean">
      Whether to automatically copy picked colors to clipboard. Default: true
    </ResponseField>
  </Accordion>
</ParamField>

<ParamField path="theme" type="string">
  Active theme name. Options: `"default"`, `"light"`, `"blue"`
</ParamField>

### Storage example

```javascript
chrome.storage.local.get(['savedColors', 'lastPickedColor'], (result) => {
  const savedColors = result.savedColors || [];
  const lastPickedColor = result.lastPickedColor || null;
  
  console.log('Saved colors:', savedColors);
  console.log('Last picked:', lastPickedColor);
});

chrome.storage.local.set({
  savedColors: ["rgb(138, 43, 226)"],
  lastPickedColor: "rgb(138, 43, 226)"
}, () => {
  console.log('Color saved!');
});
```

## Helper functions

### isTabUrlSupported

Checks if a tab URL is supported for color picking (only http/https URLs).

```javascript background.js
function isTabUrlSupported(url) {
  return !!url && /^(https?:)\/\//i.test(url);
}
```

### captureVisibleTab

Captures a screenshot of the visible area of the active tab.

```javascript background.js
function captureVisibleTab(windowId) {
  return new Promise((resolve, reject) => {
    chrome.tabs.captureVisibleTab(windowId, {
      format: 'png',
      quality: 100
    }, (imageUri) => {
      if (chrome.runtime.lastError || !imageUri) {
        reject(new Error(chrome.runtime.lastError?.message || 'Unable to capture visible tab'));
        return;
      }

      resolve(imageUri);
    });
  });
}
```
